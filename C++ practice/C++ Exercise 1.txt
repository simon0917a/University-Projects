#include <iostream>
using namespace std;
int main () {
    float Celcius;
    cout << "Enter a temperature in Celsius: ";
    cin >> Celcius;
    cout << "Temperature in Fahrenheit is: " << Celcius*9/5+32 << "\n";

}

Q1.3 (other method)

#include <iostream>
int main() {
    float Celsius;
    std::cout << "Enter a temperature in Celsius: ";
    std::cin >> Celsius;
    float Fahrenheit = (Celsius * 9/5) + 32;
    std::cout << "Temperature in Fahrenheit: " << Fahrenheit << "\n";
    return 0;
}
