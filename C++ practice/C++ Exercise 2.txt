#include <iostream>
using namespace std;
int main () {
    float coursework;
    cout << "Enter coursework marks: ";
    cin >> coursework;
    float examination;
    cout << "Enter examination marks: ";
    cin >> examination;
    float Overall = (coursework*0.4 )+ (examination*0.6);
    if (coursework >= 50 && examination >=50)
    cout << "Your overall result is " << Overall << " (Passed)\n";
    else
    cout << "Your overall result is " << Overall << " (Failed)\n";
    return 0;
}
